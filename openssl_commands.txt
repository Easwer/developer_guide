-------------------------------------------------------------------------------------------------------------------------------------------------------------
Create Root Key
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl genrsa -des3 -out rootCA.key 4096

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Create and self sign the Root Certificate
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl req -x509 -new -nodes -key rootCA.key -sha256 -days 7300 -out rootCA.crt

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Generate KEY (.key .pem .der)
-------------------------------------------------------------------------------------------------------------------------------------------------------------

RSA
openssl genrsa -des3 -out server.key 2048 (with passphrase)
openssl genrsa -out server.key 2048 (without passphrase)

DSA
openssl dsaparam -out server.key 2048 (create dsa param)
openssl gendsa -out easwer.key server.key (without passphrase)
openssl gendsa -out easwer.key server.key -aes256 (with passphrase)

ECDSA
openssl ecparam -name secp521r1 -genkey -out server.key (without passphrase)
openssl ecparam -name secp521r1 -param_enc explicit -genkey -out server.key (without passphrase)
openssl ec -in server.key -out easwer.key -aes256 (with passphrase last command + this)

For name in previous command
openssl ecparam -list_curves

EdDSA (Ed25519)
openssl genpkey -algorithm Ed25519 -out ed25519key.pem

This works only on latest openssl.
-------------------------------------------------------------------------------------------------------------------------------------------------------------
Generate openSSL CSR from old key
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl req -new -sha256 -key server.key -out server.csr

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Validate/View KEY
-------------------------------------------------------------------------------------------------------------------------------------------------------------

RSA
openssl rsa -in server.key -check

DSA
openssl dsaparam -in server.key -text -noout

ECDSA

openssl ecparam -in server.key -text -noout

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Generate openSSL new KEY and CERTIFICATE
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl req -x509 -newkey rsa:2048 -keyout server.key -out server.crt -nodes

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Generate openSSL self-signed CERTIFICATE using key and csr
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl x509 -req -sha256 -days 7300 -in server.csr -signkey server.key -out server.crt

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Generate openSSL signed CERTIFICATE from CA
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl x509 -req -in server.csr -CA rootCA.crt -CAkey rootCA.key -CAcreateserial -out server.crt -days 7300 -sha256

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Validate/View CERTIFICATE
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl x509 -text -noout -in server.crt

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Combine your key and certificate to generate PKCS (P12)
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl pkcs12 -inkey server.key -in server.crt -export -out server.p12

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Validate your P12 file
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl pkcs12 -in server.p12 -noout -info

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Convert a DER file (.crt .cer .der) to PEM
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl x509 -inform der -in server.cer -out server.pem

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Convert a PEM file to DER
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl x509 -outform der -in server.pem -out server.der

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Convert a PKCS#12 file (.pfx .p12) containing a private key and certificates to PEM
You can add -nocerts to only output the private key or add -nokeys to only output the certificates.
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl pkcs12 -in keyStore.pfx -out keyStore.pem -nodes

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Start openSSL server
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl s_server -cert cert.pem -certform PEM -key key.pem -keyform PEM -accept 4433 -www

openssl s_server -cert server.crt -key server.key -accept 4433 -www

-------------------------------------------------------------------------------------------------------------------------------------------------------------
Start openSSL client
-------------------------------------------------------------------------------------------------------------------------------------------------------------

openssl s_client -connect localhost:4433
